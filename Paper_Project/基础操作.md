# python-docx 基础操作速查

## 一键工作流：run_pipeline.py

```
小白操作步骤:

 1. 把模版 docx 放入 Templates/ 文件夹
 2. 把文本资料 docx（含图片）放入 Inputs/ 文件夹
 3. 打开 run_pipeline.py，修改 TEMPLATE_DOCX 和 CONTENT_DOCX 两个文件名
 4. 终端运行：python run_pipeline.py
 5. 得到：
    - Outputs/格式提取.md       → 核对模版格式是否完整
    - Outputs/内容提取.md       → 核对内容是否完整
    - Program/build_generated.py → 自动生成的构建脚本
    - Manuscripts/最终论文.docx  → 最终 docx

 6. 在 VSCode 中打开 build_generated.py，与 Claude 对话微调排版
 7. 微调完运行：python Program/build_generated.py
```

## 三线表

所有表格统一使用三线表格式：顶粗线（1.5pt）/ 表头下细线（0.5pt）/ 底粗线（1.5pt）/ 无竖线。

需要的额外 import：

```python
from docx.oxml import OxmlElement
from docx.oxml.ns import qn
```

### 核心实现（直接复制使用）

```python
def _remove_table_borders(table):
    """移除 tblBorders，避免 insideH=none 压制单元格级边框"""
    tbl = table._tbl
    tblPr = tbl.tblPr
    if tblPr is None:
        tblPr = OxmlElement('w:tblPr')
        tbl.insert(0, tblPr)
    for old in tblPr.findall(qn('w:tblBorders')):
        tblPr.remove(old)


def _set_tc_border(cell, top=None, bottom=None, left='nil', right='nil'):
    """为单个单元格设置四边边框，不依赖继承"""
    tcPr = cell._tc.get_or_add_tcPr()
    for old in tcPr.findall(qn('w:tcBorders')):
        tcPr.remove(old)
    tcB = OxmlElement('w:tcBorders')
    tcPr.append(tcB)

    def add_border(pos, val='nil', sz='0'):
        b = OxmlElement(f'w:{pos}')
        b.set(qn('w:val'), val)
        b.set(qn('w:sz'), str(sz))
        b.set(qn('w:space'), '0')
        b.set(qn('w:color'), '000000')
        tcB.append(b)

    if top is not None:
        add_border('top', top[0], top[1])
    else:
        add_border('top', 'nil', '0')
    if bottom is not None:
        add_border('bottom', bottom[0], bottom[1])
    else:
        add_border('bottom', 'nil', '0')
    add_border('left', left, '0')
    add_border('right', right, '0')


def three_line_table(table):
    """将表格转为三线表：第0行 top=粗 bottom=细 / 第1行 top=细 / 末行 bottom=粗 / 无竖线"""
    _remove_table_borders(table)
    nrows = len(table.rows)
    if nrows == 0:
        return
    for ri, row in enumerate(table.rows):
        for cell in row.cells:
            if ri == 0 and nrows == 1:
                _set_tc_border(cell, top=('single', '12'), bottom=('single', '12'))
            elif ri == 0:
                _set_tc_border(cell, top=('single', '12'), bottom=('single', '4'))
            elif ri == 1 and ri == nrows - 1:
                _set_tc_border(cell, top=('single', '4'), bottom=('single', '12'))
            elif ri == 1:
                _set_tc_border(cell, top=('single', '4'))
            elif ri == nrows - 1:
                _set_tc_border(cell, bottom=('single', '12'))
            else:
                _set_tc_border(cell)
```

### 配套：单元格填入

```python
def set_cell(cell, text, bold=False, size=9, alignment=WD_ALIGN_PARAGRAPH.CENTER):
    """设置单元格文本（清空旧内容后写入）"""
    cell.text = ''
    p = cell.paragraphs[0]
    p.alignment = alignment
    r = p.add_run(text)
    r.bold = bold
    r.font.size = Pt(size)
    r.font.name = '宋体'
```

### 用法

```python
table = doc.add_table(rows=5, cols=3)

# 填表头
for i, h in enumerate(['列A', '列B', '列C']):
    set_cell(table.rows[0].cells[i], h, bold=True, size=10)

# 填数据
data = [['a1', 'b1', 'c1'], ['a2', 'b2', 'c2'], ...]
for ri, row_data in enumerate(data, 1):
    for ci, text in enumerate(row_data):
        set_cell(table.rows[ri].cells[ci], text, size=9)

# ☆ 填完所有内容后再调用 ☆
three_line_table(table)
```

### 实现原理

1. `_remove_table_borders` — 完全移除 `tblPr/tblBorders`，避免 Word 的 `insideH=none` 压制单元格级边框
2. `_set_tc_border` — 为每个单元格独立写入 `tcBorders`（top/bottom/left/right），不依赖样式继承
3. `three_line_table` — 遍历全表，按行位置分配边框规则
4. 粗线 `('single', '12')` = 1.5pt，细线 `('single', '4')` = 0.5pt，无框 `'nil'`

**关键**：`three_line_table()` 必须在填完所有数据后调用，否则后续填入可能破坏边框设置。

## 交叉引用（正文 [N] ↔ 参考文献列表 可跳转）

实现学术论文中的 Ctrl+Click 跳转引用：正文蓝色上标 `[1],[2]` 点击后跳转到参考文献列表对应条目。

### 全局变量

```python
ref_bookmarks = {}  # ref_num -> bookmark_name
```

### ① 正文引用标记（B_ref + add_ref_hyperlink）

```python
def B_ref(text, ref_nums):
    """正文段落 + 末尾蓝色上标超链接引用标记，如 [1,2,3]"""
    p = doc.add_paragraph()
    J(p)
    R(p, text, size=12)
    for i, n in enumerate(ref_nums):
        if i > 0:
            R(p, ', ', size=10, superscript=True)
        add_ref_hyperlink(p, n, f'[{n}]')
    return p


def add_ref_hyperlink(paragraph, ref_num, display_text):
    """在段落中插入指向参考文献书签的锚点超链接（蓝色上标）"""
    bookmark_name = f'_Ref{ref_num}'
    ref_bookmarks[ref_num] = bookmark_name

    # w:anchor 方式 — 不需要 OPC relationship，避免 OPC 损坏
    hl = paragraph._element.makeelement(qn('w:hyperlink'), {
        qn('w:anchor'): bookmark_name,
        qn('w:history'): '1',
    })
    run_elem = paragraph._element.makeelement(qn('w:r'), {})
    rPr = paragraph._element.makeelement(qn('w:rPr'), {})

    # 上标
    v = paragraph._element.makeelement(qn('w:vertAlign'), {qn('w:val'): 'superscript'})
    rPr.append(v)
    # 字号 10pt → 20 half-pts
    sz = paragraph._element.makeelement(qn('w:sz'), {qn('w:val'): '20'})
    rPr.append(sz)
    # 蓝色
    c = paragraph._element.makeelement(qn('w:color'), {qn('w:val'): '0000FF'})
    rPr.append(c)

    run_elem.append(rPr)
    t = paragraph._element.makeelement(qn('w:t'), {})
    t.set(qn('xml:space'), 'preserve')
    t.text = display_text
    run_elem.append(t)
    hl.append(run_elem)
    paragraph._element.append(hl)
```

### ② 参考文献列表（带书签锚点）

每条参考文献必须以 `w:bookmarkStart` / `w:bookmarkEnd` 包裹，书签名与 `add_ref_hyperlink` 中的 `_Ref{N}` 一致：

```python
from docx.oxml import OxmlElement
from docx.oxml.ns import qn

refs = [
    (1, 'Weissmueller J. Alloy effects in nanostructures. Nanostructured Materials, 1993, ...'),
    (2, 'Smith C S. Grains, phases, and interfaces. Trans. Metall. Soc. AIME, 1948, ...'),
    # ...
]

for num, ref_text in refs:
    p = doc.add_paragraph()
    J(p)
    p.paragraph_format.left_indent = Cm(1.27)
    p.paragraph_format.first_line_indent = Cm(-1.27)

    # ★ 书签开始
    bookmark_name = f'_Ref{num}'
    bk_start = OxmlElement('w:bookmarkStart')
    bk_start.set(qn('w:id'), str(num))
    bk_start.set(qn('w:name'), bookmark_name)
    p._element.append(bk_start)

    R(p, f'[{num}] ', size=10)
    R(p, ref_text, size=10)

    # ★ 书签结束（id 必须与 start 一致）
    bk_end = OxmlElement('w:bookmarkEnd')
    bk_end.set(qn('w:id'), str(num))
    p._element.append(bk_end)
```

### ③ 用法

```python
# 正文中引用
B_ref('Weissmueller established the thermodynamic framework for GB energy.',
      [1])
B_ref('The Zener pinning model provides a classical kinetic stabilization mechanism.',
      [2])
B_ref('These two approaches together define the core stability landscape.',
      [1, 2])   # 多篇引用

# 参考文献列表（同②，逐条生成带书签的条目）
```

### 实现原理

| 组件 | 机制 | 目的 |
|------|------|------|
| `add_ref_hyperlink` | `w:hyperlink` 的 `w:anchor` 属性 | 跳转到同名书签，**不需要 OPC relationship**，避免 `r:id` 损坏 |
| `bookmarkStart`/`bookmarkEnd` | Word 内部书签 | 定义跳转目标，`w:name` 与 `w:anchor` 值匹配 |
| `ref_bookmarks` dict | 内存追踪 | 记录已注册的书签名，确保不重复 |
| 蓝色上标 `0000FF` + `superscript` | 视觉标记 | 与正文区分，明确可点击 |

**与外部超链接的区别**：外部 URL（`https://...`）必须用 OPC `r:id` + `RT.HYPERLINK`；内部交叉引用必须用 `w:anchor` + 书签。混用会导致 OPC 损坏或链接失效。

## 页眉与页码

python-docx 无内置页码 API，必须用 OOXML Word 域代码（`PAGE` field）实现动态页码——Word 打开文件时自动计算并渲染为实际数字，随页数变化自动对齐。

需要的额外 import：

```python
from docx.oxml import OxmlElement
from docx.oxml.ns import qn
```

### 页码（页脚居中）

```python
sec = doc.sections[0]
footer = sec.footer
footer.is_linked_to_previous = False  # 断开与前节的链接（多节文档时必须）
fp = footer.paragraphs[0]
fp.alignment = WD_ALIGN_PARAGRAPH.CENTER

run = fp.add_run()
run.font.size = Pt(10)
run.font.name = 'Times New Roman'

# 构造 Word 域代码：{ PAGE }
fld_begin = OxmlElement('w:fldChar')
fld_begin.set(qn('w:fldCharType'), 'begin')
run._element.append(fld_begin)

instr = OxmlElement('w:instrText')
instr.set(qn('xml:space'), 'preserve')
instr.text = ' PAGE '
run._element.append(instr)

fld_end = OxmlElement('w:fldChar')
fld_end.set(qn('w:fldCharType'), 'end')
run._element.append(fld_end)
```

### 页眉（论文标题）

### 页眉 + 页码组合（常见学术论文格式）

```python
sec = doc.sections[0]

# -- 页眉：标题右对齐 --
header = sec.header
header.is_linked_to_previous = False
hp = header.paragraphs[0]
hp.alignment = WD_ALIGN_PARAGRAPH.RIGHT
run = hp.add_run('Short Title')
run.font.size = Pt(9); run.font.name = 'Times New Roman'; run.italic = True

# -- 页脚：页码居中 --
footer = sec.footer
footer.is_linked_to_previous = False
fp = footer.paragraphs[0]
fp.alignment = WD_ALIGN_PARAGRAPH.CENTER
run = fp.add_run()
run.font.size = Pt(10); run.font.name = 'Times New Roman'

fld_begin = OxmlElement('w:fldChar')
fld_begin.set(qn('w:fldCharType'), 'begin')
run._element.append(fld_begin)
instr = OxmlElement('w:instrText')
instr.set(qn('xml:space'), 'preserve')
instr.text = ' PAGE '
run._element.append(instr)
fld_end = OxmlElement('w:fldChar')
fld_end.set(qn('w:fldCharType'), 'end')
run._element.append(fld_end)
```

### 实现原理

| 元素 | 作用 |
|------|------|
| `w:fldChar fldCharType="begin"` | 域代码开始标记 |
| `w:instrText` `PAGE` | Word 域指令，渲染为当前页码 |
| `w:fldChar fldCharType="end"` | 域代码结束标记 |
| `is_linked_to_previous = False` | 断开"链接到前一节"，多节文档必须设置 |

**为什么不用静态文本**：硬编码 `1` / `2` / `3` 不会随文档编辑后的页数变化而更新。Word `PAGE` 域是动态字段，每次打开文件或打印时自动计算正确页码。

**常见变体**：
- `{ PAGE }` — 当前页码
- `{ NUMPAGES }` — 总页数
- `Page { PAGE } of { NUMPAGES }` — "Page X of Y" 格式

## A4 自动分页

`build_acta_manuscript.py` 内置了 `paginate_a4(doc)` 函数，在 `doc.save()` 前自动调用，一个脚本即出最终 Word。

### 用法

```bash
python Program/build_acta_manuscript.py  # 排版 + 分页一键完成
```

### 算法

| 参数 | 值 |
|------|-----|
| 可用高度 | 25.0 cm ≈ 709 pt（比理论值多 ~1.5%，补偿 Word 渲染误差） |
| 每行字数（英文 12pt） | ~75 chars |
| 行高（12pt × 2.0 行距） | 24 pt |
| 每页行数 | ~29 行 |

遍历 `doc.element.body` 中所有 `w:p` 和 `w:tbl` 元素：
- 文本段：从 `w:rPr/w:sz` 取实际字号，文本长度 ÷ 75 估算行数，× 字号 × 行距得高度
- 图片段：从 `wp:extent` 取实际图片高度（EMU → pt）
- 表格：表头 28pt + 每数据行 20pt
- 累积到 ~698pt 时在当前位置前插入 `w:br type="page"`

### 防孤行（Widow/Orphan Control）

`paginate_a4()` 内置胶合（glue）逻辑，自动检测以下模式并阻止跨页拆散：

| 模式 | 检测规则 |
|------|---------|
| 短引介段 → 表格 | 文本 < 500 chars + 紧跟 `w:tbl` |
| 短引介段 → 图片 | 文本 < 500 chars + 紧跟含 `w:drawing` 的段落 |
| 图片 → 图注 | `w:drawing` 段 + 紧跟以 `Fig.`/`Table` 开头的图注 |
| 标题 → 正文 | 文本 < 100 chars + 高度 > 20pt（大字 = 标题） |

同时，关键段落设置了 Word 原生 `keep_with_next = True` 属性作为第二层保护。

### 内置函数签名

```python
def paginate_a4(doc):
    """Insert page breaks at A4 boundaries, directly on the Document object.
    Returns number of breaks inserted."""
```

直接传入 `doc` 对象即可，在其他 build 脚本中复用。
